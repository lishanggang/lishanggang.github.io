判断是否连接上sql数据库

发表于 2016-11-14   |   分类于 SQL   |     |  

//设置PHP操作数据库数据的编码格式

$conn->query("set names utf8");

//执行SQL语句

$sql="SELECT * FROM user";
$result=$conn->query($sql);        //执行完SQL查询语句返回一个数据集$result;

//$arr=$result->fetch_assoc(); //根据key获取数据,只能获取一条数据
// print_r($arr);

//循环获取表的全部数据,以表格输出

echo "<table border=1>";
while ($arr=$result->fetch_assoc()) { //循环获取每一天数据
    echo "<tr>
    <td>{$arr['id']}</td>
    <td>{$arr['name']}</td>
    <td>{$arr['pwd']}</td>
    </tr>";
}
echo "</table>";
